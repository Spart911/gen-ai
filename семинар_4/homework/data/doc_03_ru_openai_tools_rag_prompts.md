import requests

# Установите ваш ключ API от OpenAI

# Функция для загрузки содержимого документа и выполнения поиска
def search_in_google_docs(url, search_query):
 # Извлечение идентификатора документа из URL
 match = re.search(r'/document/d/([a-zA-Z0-9-_]+)', url)
 if match is None:
 raise ValueError('Недопустимый URL Google Docs')
 doc_id = match.group(1)

 # Загрузка содержимого документа как обычного текста
 response = requests.get(f'https://docs.google.com/document/d/{doc_id}/export?format=txt')
 response.raise_for_status()
 document_text = response.text

 # Выполнение поиска в тексте
 if search_query in document_text:
 return f"Текст '{search_query}' найден в документе:\n{document_text}"
 else:
 return f"Текст '{search_query}' не найден в документе."
 
 # Функция для выполнения диалога с пользователем с использованием ChatGPT
def chat_with_user():
 while True:
 user_input = input("Вы: ")
 if user_input.lower() == 'выход':
 break
 
 # Вызов ChatGPT для ответа на запрос пользователя
 response = openai.Completion.create(
 engine="text-davinci-002",
 prompt=f"Пользователь: {user_input}\nМашина:",
 max_tokens=50
 )
 machine_response = response.choices[0].text.strip()
 
 # Вывод ответа машины
 print(f"Машина: {machine_response}")
 
 # Проверка, содержит ли ответ машины информацию о поиске
 if "Поиск в документе:" in machine_response:
 # Извлечение запроса на поиск из ответа машины
 search_query = re.search(r'Поиск в документе: (.+)', machine_response).group(1)
 # Вызов функции поиска в документе
 search_result = search_in_google_docs(google_docs_url, search_query)
 # Вывод результата поиска
 print(search_result)
